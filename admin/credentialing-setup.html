Connect Google Sheets for Shared Access

This 5-minute setup lets you and your helper share the same credentialing data in real time. No coding required.

1

Create a Google Sheet

Go to sheets.google.com and create a new spreadsheet. Name it "Lumen Credentialing Tracker".

Add these headers in Row 1 (exactly as shown):

id | state | payer | wave | status | applied | followup | effective | revenue | notes
Tip: Paste the headers into cell A1. Each column header goes in its own cell (A1 through J1).
2

Open the Apps Script Editor

In your Google Sheet, go to Extensions > Apps Script. This opens the script editor.

Delete any existing code and paste the entire script from Step 3 below.

3

Paste This Script

Copy and paste the entire code below into the Apps Script editor:

// Lumen Credentialing Tracker — Google Apps Script Backend
// Paste this into Extensions > Apps Script in your Google Sheet

const SHEET_NAME = 'Sheet1'; // Change if you renamed the sheet tab
const COLUMNS = ['id', 'state', 'payer', 'wave', 'status', 'applied', 'followup', 'effective', 'revenue', 'notes'];

function doGet(e) {
  const action = e.parameter.action || 'getAll';

  if (action === 'getAll') {
    return jsonResponse({ success: true, data: getAllRows() });
  }

  return jsonResponse({ success: false, error: 'Unknown action' });
}

function doPost(e) {
  try {
    const body = JSON.parse(e.postData.contents);
    const action = body.action;
    const entry = body.entry;

    if (action === 'add') {
      addRow(entry);
      return jsonResponse({ success: true, data: getAllRows() });
    }

    if (action === 'update') {
      updateRow(entry);
      return jsonResponse({ success: true, data: getAllRows() });
    }

    if (action === 'delete') {
      deleteRow(entry.id);
      return jsonResponse({ success: true, data: getAllRows() });
    }

    if (action === 'bulkAdd') {
      // Replace all data
      const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
      const lastRow = sheet.getLastRow();
      if (lastRow > 1) {
        sheet.getRange(2, 1, lastRow - 1, COLUMNS.length).clearContent();
      }
      const allEntries = Array.isArray(entry) ? entry : body.entry;
      allEntries.forEach(function(e) { addRow(e); });
      return jsonResponse({ success: true, data: getAllRows() });
    }

    return jsonResponse({ success: false, error: 'Unknown action: ' + action });
  } catch (err) {
    return jsonResponse({ success: false, error: err.toString() });
  }
}

function getAllRows() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  if (lastRow <= 1) return [];

  const data = sheet.getRange(2, 1, lastRow - 1, COLUMNS.length).getValues();
  return data
    .filter(function(row) { return row[0] !== ''; })
    .map(function(row) {
      var obj = {};
      COLUMNS.forEach(function(col, i) {
        obj[col] = row[i] !== '' ? String(row[i]) : '';
      });
      return obj;
    });
}

function addRow(entry) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const row = COLUMNS.map(function(col) { return entry[col] || ''; });
  sheet.appendRow(row);
}

function updateRow(entry) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  if (lastRow <= 1) return;

  const data = sheet.getRange(2, 1, lastRow - 1, 1).getValues();
  for (var i = 0; i < data.length; i++) {
    if (data[i][0] === entry.id) {
      const row = COLUMNS.map(function(col) { return entry[col] || ''; });
      sheet.getRange(i + 2, 1, 1, COLUMNS.length).setValues([row]);
      return;
    }
  }
}

function deleteRow(id) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  if (lastRow <= 1) return;

  const data = sheet.getRange(2, 1, lastRow - 1, 1).getValues();
  for (var i = data.length - 1; i >= 0; i--) {
    if (data[i][0] === id) {
      sheet.deleteRow(i + 2);
      return;
    }
  }
}

function jsonResponse(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
4

Deploy as Web App

  • Click Deploy > New deployment
  • Click the gear icon next to "Select type" and choose Web app
  • Set "Execute as" to Me
  • Set "Who has access" to Anyone
  • Click Deploy
  • Authorize when prompted (click "Advanced" > "Go to project" if needed)
  • Copy the Web app URL — it looks like:
    https://script.google.com/macros/s/AKfyc.../exec
Important: "Anyone" means anyone with the URL can access the data. The tracker's access code provides the security layer. Don't share the Apps Script URL publicly.
5

Connect to the Tracker

Open the credentialing tracker HTML file and find this line near the top of the <script> section:

APPS_SCRIPT_URL: ''

Paste your web app URL between the quotes:

APPS_SCRIPT_URL: 'https://script.google.com/macros/s/AKfyc.../exec'

Save the file, commit, and push to GitHub. The tracker will now read/write from your Google Sheet.

6

Share with Your Helper

Send your helper two things:

  • Tracker URL: https://lumen-psychiatry-group.example/admin/credentialing-tracker
  • Access code: ennhealth2026 (or whatever you set in CONFIG.ACCESS_CODE)

They can open the link, enter the code, and start adding/updating applications. All changes sync to the same Google Sheet in real time.
